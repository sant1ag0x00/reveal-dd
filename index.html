Syntax


    # dd
    dd if=image.iso of=/dev/sdb bs=4M
    # cat
    cat image.iso >/dev/sdb

    # dd
    dd if=/dev/zero of=image.iso bs=4MB count=25
    # head
    head -c 100MB /dev/zero >image.iso
